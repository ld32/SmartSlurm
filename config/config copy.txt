adjustPartition() {
    [ -z "$1" ] && { echo -e "Usage: $0 <mem (M)> <time (min)> <partition>\nReturn: \$mem, \$min and \$partition"; return; }
    mem=$1
    min=$2
    partitionO=$3 
    for i in {11..20}; do
        partition=partition${i}Name
        partition=${!partition}
        
        [ -z "$partition" ] && break
        if [[ "$partition" ==  "$partitionO" ]]; then
            memLimit=partition${i}MemLimit
            memLimit=${!memLimit}
            [ "$mem" -gt "$memLimit" ] && mem="$memLimit" 
    
            timeLimit=partition${i}TimeLimit
            timeLimit=${!timeLimit}
            [ "$min" -gt "$timeLimit" ] && min="$timeLimit"
            return
        else 
            partition=$partitionO
        fi
    done
    
    for i in {1..10}; do
        partition=partition${i}Name
        [ -z ""${!partition}"" ] && continue
        partitionTime=partition${i}TimeLimit
        if [ $min -le  "${!partitionTime}" ]; then
            partition=${!partition}
            memLimit=partition${i}MemLimit
            memLimit=${!memLimit}
            [ "$mem" -gt "$memLimit" ] && mem="$memLimit" 
            timeLimit=partition${i}TimeLimit
            timeLimit=${!timeLimit}
            [ "$min" -gt "$timeLimit" ] && min="$timeLimit"
            break
        else 
            partition=$partitionO
        fi
    done
}
export -f adjustPartition
